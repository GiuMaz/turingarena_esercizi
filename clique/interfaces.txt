interface light_coin {


    function isClique(int n,int[] C, int[][] AdjG, int k) -> int;
    function isIndependentSet(int n, int[] S, int[][] AdjH, int h) -> int;

    callback place(int coin, int position)
    {
        output coin, position;
    }

    main {
        var int n, k;
        var int[][] AdjG;
        var int[] C,

        input n;
        alloc AdjG : n;
        for(i: n) {
            alloc AdjG[i] : n;
        }

        for(i: n) {
            for(j: n) {
                input AdjG[i][j];
            }
        }

        input k;
        alloc C : k;
        for (i: k) {
            input C[i];
        }

        var int ans;
        call isClique(n,C,AdjG,k) -> ans;
        output ans;
    }
}
