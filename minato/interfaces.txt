interface minato {

    var int N;
    var int M;
    var int[][] mine_field;

    function find_number_of_paths() -> int;

    main {

        input N;
        input M;

        alloc mine_field : 1..N;
        for(i: 1..N) {
            alloc mine_field[i] : 1..M;
        }

        for(i: 1..N) {
            for(j: 1..M) {
                input mine_field[i][j];
            }
        }

        var int ans;
        call find_number_of_paths() -> ans;

        output ans;
    }
}
