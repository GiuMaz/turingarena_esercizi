interface minato {

    var int N, M;
    var int[][] mine_field;

    function find_number_of_paths() -> int;

    main {
        input N, M;

        alloc mine_field : N;
        for(i: N) {
            alloc mine_field[i] : M;
        }

        for(i: N) {
            for(j: M) {
                input mine_field[i][j];
            }
        }

        var int ans;
        call find_number_of_paths() -> ans;

        output ans;
    }
}
