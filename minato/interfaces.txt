interface minato {

    var int N;
    var int M;

    function find_number_of_paths() -> int;

    callback is_safe(int row, int column) -> int
    {
        output row, column;
        flush;
        var int ans;
        input ans;
        return ans;
    }

    main {

        input N;
        input M;

        var int ans;

        call find_number_of_paths() -> ans;

        output ans;
    }
}
